#!/bin/bash
set -u

# Verificar privilegios de root
if [ "$(id -u)" -ne 0 ]; then
    echo "‚ùå Este script debe ejecutarse como root."
    exit 1
fi

# Verificar si firewalld est√° instalado
if ! command -v firewall-cmd &>/dev/null; then
    echo "‚ùå firewalld no est√° instalado."
    exit 1
fi

# Iniciar firewalld si no est√° activo
if ! systemctl is-active --quiet firewalld; then
    echo "‚öôÔ∏è Iniciando firewalld..."
    systemctl enable --now firewalld || {
        echo "‚ùå No se pudo iniciar firewalld."
        exit 1
    }
fi

# Obtener zona activa
ZONA_ACTIVA=$(firewall-cmd --get-active-zones | awk 'NR==1 {print $1}')
ZONA_ACTIVA=${ZONA_ACTIVA:-public}
echo "üåê Zona activa detectada: $ZONA_ACTIVA"

# Definir listas de servicios
SERVICIOS_OK=("ssh" "http" "https")
SERVICIOS_INSEGUROS=("ftp" "telnet" "samba" "smtp")

echo ""
echo "üßπ Eliminando servicios inseguros..."
for svc in "${SERVICIOS_INSEGUROS[@]}"; do
    if firewall-cmd --permanent --zone="$ZONA_ACTIVA" --query-service="$svc" &>/dev/null; then
        firewall-cmd --permanent --zone="$ZONA_ACTIVA" --remove-service="$svc"
        echo "  üîª Eliminado: $svc"
    fi
done

echo ""
echo "‚úÖ Asegurando servicios esenciales..."
for svc in "${SERVICIOS_OK[@]}"; do
    if ! firewall-cmd --permanent --zone="$ZONA_ACTIVA" --query-service="$svc" &>/dev/null; then
        firewall-cmd --permanent --zone="$ZONA_ACTIVA" --add-service="$svc"
        echo "  ‚ûï Permitido: $svc"
    fi
done

# Recargar configuraci√≥n
echo ""
echo "‚ôªÔ∏è Recargando configuraci√≥n del firewall..."
firewall-cmd --reload

# Mostrar configuraci√≥n final
echo ""
echo "üéØ Configuraci√≥n final de servicios permitidos en zona '$ZONA_ACTIVA':"
firewall-cmd --zone="$ZONA_ACTIVA" --list-services

#!/bin/bash
set -u

ARCHIVOS_SENSIBLES=(
    "/etc/shadow:600"
    "/etc/gshadow:600"
    "/etc/passwd:644"
    "/etc/group:644"
    "/root:700:d"
)

echo "Ajustando permisos cr√≠ticos..."

for item_perm in "${ARCHIVOS_SENSIBLES[@]}"; do
    IFS=':' read -ra permisos <<< "$item_perm"
    item="${permisos[0]}"
    perm_deseado="${permisos[1]}"
    tipo="${permisos[2]:-f}"

    echo "Ajustando $item a $perm_deseado..."
    sudo chmod "$perm_deseado" "$item" || echo "ERROR ajustando $item"

    case "$tipo" in
        d) ls -ld "$item" ;;
        f) ls -l "$item" ;;
        *) echo "ERROR: tipo de archivo desconocido para $item" ;;
    esac
done

#!/bin/bash

# Detectar autom√°ticamente el gestor de paquetes
detectar_gestor_paquetes() {
    if command -v dnf &>/dev/null; then
        echo "dnf install -y"
    elif command -v yum &>/dev/null; then
        echo "yum install -y"
    elif command -v apt &>/dev/null; then
        echo "apt install -y"
    elif command -v pacman &>/dev/null; then
        echo "pacman -Sy --noconfirm"
    else
        echo ""
    fi
}

# Verifica si un comando existe, y si no, intenta instalar el paquete
# Uso: asegurar_comando nombre_comando [nombre_paquete]
asegurar_comando() {
    local comando="$1"
    local paquete="${2:-$1}"  # Usa el mismo nombre si no se especifica

    if ! command -v "$comando" &>/dev/null; then
        echo "üîç $comando no est√° instalado. Intentando instalar..."
        local instalador=$(detectar_gestor_paquetes)
        if [ -n "$instalador" ]; then
            echo "üí° Ejecutando: sudo $instalador $paquete"
            sudo $instalador "$paquete" || {
                echo "‚ùå Fall√≥ la instalaci√≥n de $paquete"
                exit 1
            }
        else
            echo "‚ùå No se pudo detectar un gestor de paquetes v√°lido."
            exit 1
        fi
    else
        echo "‚úÖ $comando ya est√° instalado."
    fi
}
